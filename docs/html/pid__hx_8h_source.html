93 int Compute() {
94 if (!inAuto) return 0; // No computation if not in automatic mode
95
96 unsigned long now = millis();
97 int timeChange = now - lastTime;
98
99 if (timeChange >= sampleTime) {
100 // Calculate error and derivative
101 float error = setpoint - input;
102 float dInput = input - lastInput;
103
104 // Update the integral term
105 outputSum += (ki * error);
106
107 // Adjust for proportional on measurement (if enabled)
108 if (pOnM) outputSum -= pOnMKp * dInput;
109
110 // Clamp the output sum to prevent windup
111 if (outputSum > outMax) outputSum = outMax;
112 else if (outputSum < outMin) outputSum = outMin;
113
114 // Compute the final PID output
115 float proportional = pOnE ? (pOnEKp * error) : 0;
116 float derivative = -kd * dInput;
117
118 output = proportional + outputSum + derivative;
119
120 // Clamp the output to specified limits
121 if (output > outMax) output = outMax;
122 else if (output < outMin) output = outMin;
123
124 // Save state for the next iteration
125 lastInput = input;
126 lastTime = now;
127
128 return 1; // Indicate that new values were computed
129 }
130
131 return 0; // No new computation
132 }

140 void SetTunings(float Kp, float Ki, float Kd) {
141 if (Kp < 0 || Ki < 0 || Kd < 0) return;

143 float sampleTimeInSec = ((float)sampleTime) / 1000.0f;
144 kp = Kp;
